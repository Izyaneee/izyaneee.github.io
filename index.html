// Create an array to represent the playing card deck values const deck = [ 2, 3, 4, 5, 6, 7, 8, 9, 10, 10, 10, 10, // Numbers and 10s 11, 11, 11, 11 // Four Aces ]; // Function to simulate dealing a card from the deck function dealCard() { const randomIndex = Math.floor(Math.random() * deck.length); return deck.splice(randomIndex, 1)[0]; } // Function to play as the dealer function playDealer() { let total = 0; // Deal the first two cards const card1 = dealCard(); const card2 = dealCard(); total = card1 + card2; console.log(`Dealer's first two cards: ${card1} and ${card2}`); // Check for Blackjack if (total === 21 && (card1 === 11 || card2 === 11)) { console.log("Dealer has Blackjack!"); return; } // Play according to dealer rules while (total < 17) { const newCard = dealCard(); console.log(`Dealer hits and receives a ${newCard}`); total += newCard; } console.log(`Dealer's total: ${total}`); if (total > 21) { console.log("Dealer busted!"); } else { console.log("Dealer stays."); } } // Run the dealer simulation playDealer();
