function calculateCircumference() { // Get the current hour from the Date object const currentHour = new Date().getHours(); // The radius of the circle is the current hour const radius = currentHour; // Calculate the circumference const circumference = 2 * Math.PI * radius; // Round the circumference to three decimal places const roundedCircumference = circumference.toFixed(3); // Display the result console.log(`The circumference of the circle with a radius of ${radius} (current hour) is approximately ${roundedCircumference}`); } calculateCircumference();
